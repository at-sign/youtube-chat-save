function escapeHtml(text) {
  const map = {'&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#039;'};
  return text.replace(/[&<>"']/g, m => map[m]);
}

function getMessageTimestamp() {
  const now = new Date();
  return now.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit', second: '2-digit', hour12: true });
}

function parseMessage(messageElement) {
try {
const messageId = messageElement.getAttribute('id');

if (!messageId) {
  return null;
}

if (processedMessageIds.has(messageId)) {
  return null;
}

const authorNameEl = messageElement.querySelector('#author-name');
let altAuthorEl = null;

if (!authorNameEl) {
  altAuthorEl = messageElement.querySelector('[id*="author"]') ||
                messageElement.querySelector('span[class*="author"]') ||
                messageElement.querySelector('yt-formatted-string');
  if (!altAuthorEl) {
    return null;
  }
}

const username = (authorNameEl || altAuthorEl).textContent.trim();

const messageEl = messageElement.querySelector('#message');
let altMessageEl = null;

if (!messageEl) {
  altMessageEl = messageElement.querySelector('[id*="message"]') ||
                 messageElement.querySelector('yt-formatted-string:last-child') ||
                 messageElement.querySelector('span[class*="message"]');
  if (!altMessageEl) {
    return null;
  }
}

const messageTextEl = messageEl || altMessageEl;
const messageHtml = messageTextEl.innerHTML;

const timestamp = getMessageTimestamp();

const avatarImg = messageElement.querySelector('img#img');
let avatarSrc = '';
if (avatarImg) {
  avatarSrc = avatarImg.getAttribute('src') || '';
}

processedMessageIds.add(messageId);

const result = {
  username: username,
  timestamp: timestamp,
  messageHtml: messageHtml || '',
  avatar: avatarSrc
};


return result;
} catch (e) {
console.error('[Chat Monitor] Parse error:', e);
return null;
}
}

console.log('[Chat Monitor] Saved ' + capturedMessages.length + ' messages as ' + fileName);

}



function scanForMessages() {
try {
console.log('[Chat Monitor] Starting scan');

let chatContainer = null;

try {
  const chatFrame = document.querySelector('iframe#chatframe');
  if (chatFrame && chatFrame.contentDocument) {
    chatContainer = chatFrame.contentDocument.querySelector('div#chat');
    if (chatContainer) {
      console.log('[Chat Monitor] Found chat container in iframe');
    }
  }
} catch (e) {
  console.log('[Chat Monitor] Cannot access iframe (cross-origin)');
}

if (!chatContainer) {
  chatContainer = document.querySelector('div#contents.style-scope.yt-live-chat-app');
  if (chatContainer) {
    console.log('[Chat Monitor] Found chat container on main page');
  }
}

if (!chatContainer) {
  chatContainer = document.querySelector('yt-live-chat-item-list-renderer');
  if (chatContainer) {
    console.log('[Chat Monitor] Found chat container (yt-live-chat-item-list-renderer)');
  }
}

if (!chatContainer) {
  console.log('[Chat Monitor] Chat container not found');
  const liveChat = document.querySelector('yt-live-chat-app');
  if (liveChat) {
    console.log('[Chat Monitor] Found yt-live-chat-app element');
  }
  const messages = document.querySelectorAll('yt-live-chat-text-message-renderer');
  console.log('[Chat Monitor] Found ' + messages.length + ' message elements directly');
  updateLog(messages.length, 0, capturedMessages.length);
  return;
}

const messageElements = chatContainer.querySelectorAll('yt-live-chat-text-message-renderer');
console.log('[Chat Monitor] Found ' + messageElements.length + ' message elements');

let newMessagesCount = 0;
messageElements.forEach((messageEl) => {
  const parsedMessage = parseMessage(messageEl);
  if (parsedMessage) {
    capturedMessages.push(parsedMessage);
    newMessagesCount++;
  }
});

console.log('[Chat Monitor] New messages: ' + newMessagesCount + ', Total: ' + capturedMessages.length);
updateLog(messageElements.length, newMessagesCount, capturedMessages.length);
updateMonitoringBox();

} catch (e) {
console.error('[Chat Monitor] Scan error:', e);
}
}
